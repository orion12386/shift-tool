    <script>
        function generateSchedule() {
            const rawData = document.getElementById('excelData').value.trim();
            const code = document.getElementById('shiftCode').value.trim().toUpperCase();
            const resultDiv = document.getElementById('result');

            if (!rawData || !code) {
                alert('請確實貼上資料並輸入代號！');
                return;
            }

            // 解析貼上的 Excel TSV 資料
            const rows = rawData.split('\n').map(row => row.split('\t'));
            const headers = rows[0];
            
            // 動態找出時間欄位與對應的席位欄位 (支援右側第二區塊的 KHCO)
            let blocks = [];
            let currentBlock = null;

            for (let j = 0; j < headers.length; j++) {
                const header = headers[j] ? headers[j].trim() : "";
                if (header === "開始時間") {
                    if (currentBlock) blocks.push(currentBlock);
                    currentBlock = { startCol: j, endCol: j + 1, seatCols: [] };
                    j++; // 跳過下一個預期的 "結束時間"
                } else if (header && header !== "結束時間" && currentBlock) {
                    currentBlock.seatCols.push({ name: header, colIndex: j });
                }
            }
            if (currentBlock) blocks.push(currentBlock);

            let schedule = [];

            // 掃描每一列尋找代號
            for (let i = 1; i < rows.length; i++) {
                const cols = rows[i];
                blocks.forEach(block => {
                    if (cols.length > block.startCol && cols.length > block.endCol) {
                        const startTime = cols[block.startCol] ? cols[block.startCol].trim() : "";
                        const endTime = cols[block.endCol] ? cols[block.endCol].trim() : "";

                        if (startTime && endTime) {
                            block.seatCols.forEach(seat => {
                                if (cols.length > seat.colIndex) {
                                    const cellValue = cols[seat.colIndex] ? cols[seat.colIndex].trim().toUpperCase() : "";
                                    if (cellValue === code) {
                                        schedule.push({ start: startTime, end: endTime, seat: seat.name });
                                    }
                                }
                            });
                        }
                    }
                });
            }

            // 依照開始時間排序 (避免左右兩塊表格的時間交錯)
            schedule.sort((a, b) => a.start.localeCompare(b.start));

            // 合併同席位且連續的時間
            let mergedSchedule = [];
            if (schedule.length > 0) {
                let currentSlot = { ...schedule[0] };

                for (let i = 1; i < schedule.length; i++) {
                    const nextSlot = schedule[i];
                    if (currentSlot.end === nextSlot.start && currentSlot.seat === nextSlot.seat) {
                        currentSlot.end = nextSlot.end; // 延長時段
                    } else {
                        mergedSchedule.push(currentSlot);
                        currentSlot = { ...nextSlot };
                    }
                }
                mergedSchedule.push(currentSlot);
            }

            // 顯示結果
            if (mergedSchedule.length === 0) {
                resultDiv.innerHTML = `<div class="card empty-msg">找不到代號 <strong>[ ${code} ]</strong> 的排班資訊。</div>`;
                return;
            }

            let html = `<h3>[ ${code} ] 今日席位行程</h3>`;
            mergedSchedule.forEach(slot => {
                html += `
                    <div class="schedule-item">
                        <span class="time">${slot.start} - ${slot.end}</span>
                        <span class="seat">${slot.seat}</span>
                    </div>
                `;
            });
            resultDiv.innerHTML = html;
        }
    </script>
